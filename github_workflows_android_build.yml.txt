name: Build Android APK

on:
  push:
    branches:
      - main
      - master
  workflow_dispatch:

jobs:
  build:
    runs-on: ubuntu-latest

    steps:
      - name: Checkout Code
        uses: actions/checkout@v4

      - name: Setup Node.js
        uses: actions/setup-node@v4
        with:
          node-version: '20'
          cache: 'npm'

      - name: Install Dependencies
        run: npm install --legacy-peer-deps

      - name: Build Web App
        run: npm run build
        env:
          CI: false
          # Placeholder key allows build to pass without exposing real credentials
          API_KEY: "placeholder_key_for_build"

      - name: Setup Java
        uses: actions/setup-java@v4
        with:
          distribution: 'zulu'
          java-version: '17'

      - name: Initialize Capacitor Android
        run: |
          # Add android platform if not present (since it is gitignored)
          if [ ! -d "android" ]; then
            npx cap add android
          fi
          # Sync web assets and config to the android project
          npx cap sync android

      - name: Patch Android Build for Java 17
        run: |
          # The generated template might request Java 21, but Capacitor 6 works best with 17.
          # We patch the build files to enforce Java 17 compatibility.
          find android -name "build.gradle" -type f -exec sed -i 's/JavaVersion.VERSION_21/JavaVersion.VERSION_17/g' {} +
          find android -name "build.gradle" -type f -exec sed -i 's/sourceCompatibility = 21/sourceCompatibility = 17/g' {} +
          find android -name "build.gradle" -type f -exec sed -i 's/targetCompatibility = 21/targetCompatibility = 17/g' {} +

      - name: Build APK with Gradle
        working-directory: android
        run: |
          chmod +x gradlew
          ./gradlew assembleDebug --stacktrace

      - name: Upload APK
        uses: actions/upload-artifact@v4
        with:
          name: Dolphin-Debug.apk
          path: android/app/build/outputs/apk/debug/app-debug.apk
